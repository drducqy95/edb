look for=tìm kiếm
look after=chăm sóc
turn on=bật
turn off=tắt
get out=cút ra ngoài
run away=bỏ chạy

# ========================================================
# COLLOCATIONS (CỤM TỪ CỐ ĐỊNH) - Động từ + Danh từ
# ========================================================

# --- MAKE ---
make a decision=ra quyết định
make a mistake=phạm sai lầm
make sense=có ý nghĩa
make money=kiếm tiền
make friends=kết bạn
make sure=đảm bảo
make an effort=nỗ lực

# --- DO ---
do business=kinh doanh
do homework=làm bài tập
do a favor=giúp đỡ
do research=nghiên cứu
do harm=gây hại

# --- TAKE ---
take a break=nghỉ ngơi
take care of=chăm sóc
take advantage of=tận dụng
take place=diễn ra
take a seat=ngồi xuống
take a look=nhìn qua

# --- HAVE ---
have a good time=vui vẻ
have a problem=gặp vấn đề
have trouble=gặp rắc rối
have an idea=có ý tưởng

# --- PAY ---
pay attention=chú ý
pay a visit=ghé thăm
pay cash=trả tiền mặt

# --- CATCH/KEEP ---
catch a cold=bị cảm lạnh
catch fire=bắt lửa
keep in touch=giữ liên lạc
keep a promise=giữ lời hứa
